    private void btnPedidoRealizadoActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnPedidoRealizadoActionPerformed
        if (lstPedidos.getSelectedIndex() >= 0) {
            viaje.pedidoRealizado(lstPedidos.getSelectedIndex());
            actualizarLista();
            if (viaje.getListaPedidos().isEmpty()) {
                comunicacion.getVentanaInicioT().notificar(false);
            }
        }
    }//GEN-LAST:event_btnPedidoRealizadoActionPerformed

    private void actualizarLista() {
     
            lstPedidos.setEnabled(true);
            lstPedidos.setListData(viaje.getListaPedidos().toArray());
            lstPedidos.setSelectedIndex(0);

        if (!viaje.getListaPedidos().isEmpty()) {
            comunicacion.getVentanaInicioT().notificar(true);
        }
    }
