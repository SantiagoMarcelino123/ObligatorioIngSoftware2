    private void slVolumenStateChanged(javax.swing.event.ChangeEvent evt) {//GEN-FIRST:event_slVolumenStateChanged
        int volumen = slVolumen.getValue();
        if (volumen == 0) {
            lblVolumen.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rvol1.png")));
        } else if (volumen <= 35) {
            lblVolumen.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rvol2.png")));
        } else if (volumen >= 70) {
            lblVolumen.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rvol3.png")));
        } else if (volumen == 100) {
            lblVolumen.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rvol4.png")));
        }
    }//GEN-LAST:event_slVolumenStateChanged

    private void btnPlayMousePressed(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_btnPlayMousePressed
        if (musica.estaReproduciendo()) {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/RstopP.png")));
            musica.parar();
            timer.stop();
            timepo = 0;
            progressBar.setValue(0);
        } else {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/RplayP.png")));
            musica.reproducir();
            timer = new Timer(150, this);
            timer.start();
        }
    }//GEN-LAST:event_btnPlayMousePressed

    private void btnPlayMouseReleased(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_btnPlayMouseReleased
        if (musica.estaReproduciendo()) {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rstop.png")));
        } else {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rplay.png")));
        }
    }//GEN-LAST:event_btnPlayMouseReleased

    private void btnAntActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnAntActionPerformed
        if (musica.estaReproduciendo()) {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rstop.png")));
        } else {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rplay.png")));
        }
        
        if (musica.estaReproduciendo()) {
            timer.stop();
            progressBar.setValue(0);
            timepo = 0;
        }
        musica.anterior();
        if (lstCanciones.getSelectedIndex() == 0) {
            lstCanciones.setSelectedIndex(2);
        } else {
            lstCanciones.setSelectedIndex(lstCanciones.getSelectedIndex()- 1);
        }
    }//GEN-LAST:event_btnAntActionPerformed

    private void btnSigActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_btnSigActionPerformed
        if (musica.estaReproduciendo()) {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rstop.png")));
        } else {
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rplay.png")));
        }
        
        if (musica.estaReproduciendo()) {
            timer.stop();
            progressBar.setValue(0);
        timepo = 0;
        }
        musica.siguiente();
        if (lstCanciones.getSelectedIndex() == 2) {
            lstCanciones.setSelectedIndex(0);
        } else {
            lstCanciones.setSelectedIndex(lstCanciones.getSelectedIndex() + 1);
        }
    }//GEN-LAST:event_btnSigActionPerformed

    @Override
    public void update(Observable o, Object arg) {
        if (comunicacion.getBPanel()) {
            this.setVisible(false);
            comunicacion.getVentanaInicioP().mostrarSubMenu(comunicacion.getPanel());
        }
    }
    private static final Logger LOG = getLogger(MenuReproducirMusica.class.getName());

    @Override
    public void actionPerformed(ActionEvent e) {
        if (timepo == 100) {
            timer.stop();
            timepo = 0;
            if (musica.estaReproduciendo()) {
                musica.parar();
                progressBar.setValue(0);
            }
            btnPlay.setIcon(new javax.swing.ImageIcon(getClass().getResource("/vuelouy/image/Rplay.png")));
        } else {
            timepo++;
            progressBar.setValue(timepo);
        }
    }
