    ImageIcon imgDespArrib = new ImageIcon(getClass().getResource("/vuelouy/image/cinto.gif"));
    ImageIcon imgTurb = new ImageIcon(getClass().getResource("/vuelouy/image/cuidado.gif"));
    ImageIcon imgBienv = new ImageIcon(getClass().getResource("/vuelouy/image/saludo.gif"));

    @Override
    public void update(Observable o, Object arg) {

        if (comunicacion.getBMensaje()) {
            if (!comunicacion.getMensaje().equals("Desaparecer")) {
                cargarPanel(comunicacion.getMensaje());
                this.setVisible(true);
            } else {
                this.setVisible(false);
                comunicacion.setbMensaje(false);
            }
        }
    }

    /**
     *
     * @param opcion
     */
    public void cargarPanel(String opcion) {

        switch (opcion) {
            case "Despegando":
                lblImagen.setIcon(imgDespArrib);
                if (comunicacion.getIdioma().equals("espa√±ol")) {
                    lblTexto.setText("Por favor abrochen sus cinturones "
                            + ", estamos despegando");
                } else {
                    lblTexto.setText("Please fasten your seatbelts, "
                            + "we are taking off");
                }
                break;
            case "Arribando":
                lblImagen.setIcon(imgDespArrib);
                if (comunicacion.getIdioma().equals("ingles")) {
                    lblTexto.setText("Please fasten your seatbelts, "
                            + "we are landing");
                } else {
                    lblTexto.setText("Por favor abrochen sus cinturones,"
                            + " estamos aterrizando");
                }
                break;
            case "Turbulencia":
                lblImagen.setIcon(imgTurb);
                if (comunicacion.getIdioma().equals("ingles")) {
                    lblTexto.setText("Take care of your belongings, "
                            + "we are entering a turbulence zone");
                } else {
                    lblTexto.setText("Cuidado con sus pertenecias, entramos en zona de turbulencia");
                }
                break;
            case "Bienvenida":
                lblImagen.setIcon(imgBienv);
                if (comunicacion.getIdioma().equals("ingles")) {
                    lblTexto.setText("Welcome. The captain wishes you "
                            + "a good trip");
                } else {
                    lblTexto.setText("Bienvenido/a. El capit√°n les desea"
                            + " un buen viaje");
                }
                break;
            default:
                break;
        }
    }
